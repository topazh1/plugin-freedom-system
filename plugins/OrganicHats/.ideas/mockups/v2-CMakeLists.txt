# ===== WebView UI Resources =====
# Add this section to your plugin's CMakeLists.txt

# Create binary data from UI files
juce_add_binary_data(OrganicHats_UIResources
    SOURCES
        Source/ui/public/index.html
)

# Link UI resources to plugin target
target_link_libraries(OrganicHats
    PRIVATE
        OrganicHats_UIResources
        juce::juce_gui_extra
        juce::juce_recommended_config_flags
        juce::juce_recommended_lto_flags
        juce::juce_recommended_warning_flags
)

# Enable WebView support
target_compile_definitions(OrganicHats
    PUBLIC
        JUCE_WEB_BROWSER=1
        JUCE_USE_CURL=0
)
